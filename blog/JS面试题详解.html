原型链

  • 构造函数

    function Foo(name, age) {
    this.name = name;
    this.age = age;
    this.class = 'class-1';
    // return this; // 默认有这行
    // Foo开头大写，默认认知就是构造函数
    }
var f = new Foo('zhangsan',20);
// var fn1 = new Foo('list',22) //创建多个对象
  • 构造函数-扩展
    • var = a {} 其实是 var a = new Object()的语法糖
    • var = a [] 其实是 var a = new Array()的语法糖
    • function Foo(){…} 其实是 var Foo = new Function(…)
    • 使用instanceof判断一个函数是否是一个变量的构造函数
      • 判断一个变量是否为’数组’：变量 instanceof Array

  • 5个常见原型规则

    1. 所有的引用类型（数组、对象、函数），都具有对象特性，即可自由扩展属性（除了”null”意外）

      var obj = {}; obj.a = 100;
var arr = []; arr.a = 100;
function fn () {};
fn.a = 100;

    2. 所有的引用类型（数组、对象、函数），都有一个_proto_（隐式原型）属性，属性值是一个普通的对象

      console.log(obj.__proto__);
console.log(arr.__proto__);
consolo.log(fn.__proto__);

    3. 所有的函数，都有一个prototype（显式原型）属性，属性值是一个普通的对象

      console.log(fn.prototype);

    4. 所有的引用类型（数组、对象、函数），__proto__（隐式原型）属性值指向（完全等）它的构造函数的prototype（显式原型）属性值

      console.log(obj.__proto__ === Object.prototype);

    5. 当试图得到一个对象的某个属性时，如果这个对象本身没有这个属性，那么会去它的__proto__（隐式原型）（既它的构造函数的prototype（显式原型））中寻找

      //构造函数
funciton Foo(name,age) {
    this.name = name;
    }
Foo.prototype.alertName = function () {
    alert(this.name);
    };
//创建示例
var f = new Foo('zhangsan');
f.printName = function () {
    console.log(this.name);
    };
//测试
f.printName();
f.alertName();
f.toString(); //要去 f.__proto__.__proto___（下图所示）


  • 规则的补充两点：

    1、循环对象自身的属性——`f.hasOwnProperty()`
2、this永远指向对象本身

  • 原型链—instanceof

    • f instanceof Foo 的判断逻辑是：
    • f 的__proto__一层一层往上，能否对应到Foo.prototype
    • 再试着判断f instanceof Object

    • 原型链示例：

      // 定义一个构造函数
 function Elem(id) {
     this.elem= document.getElementById(id);
 }
 // 扩展原型
 Elem.prototype.html = function (val) {
     var elem = this.elem;
     if (val) {
         elem.innerHTML = val;
         return this; // 链式操作
     } else {
         return elem.innerHTML
     }
 };
 // 扩展方法(事件绑定）
 Elem.prototype.on = function (type, fn) {
     var elem = this.elem;
     elem.addEventListener(type, fn);
 };

 var div1 = new Elem('head');
 console.log(div1.html());
div1.html('<p>hello world</p>');
div1.on('click',function () {
    alert('clicked')
})    ;
//以上是一个封装DOM查询的例子

作用域和闭包

  • 知识点总结
    • 执行上下文
    • this
    • 作用域
    • 作用域链
    • 闭包

  • 执行上下文

    • 范围：一段<script>或者一个函数（生成一个执行上下文）
    • 全局：变量定义、函数声明 （针对一段<script>，执行之前把其拿出来）
    • 函数：变量定义、函数声明、this、*arguments （针对一个函数执行之前）
      • 解释：arguments
        • 使用特殊对象 arguments，开发者无需明确指出参数名
        • 用 arguments.length 检测函数的参数个数
        • 用 arguments 对象判断传递给函数的参数个数，可模拟函数重载

    • PS：注意函数表达式和函数声明的区别

      function fn(){};     // 函数声明
var fn = function(){};        // 函数表达式

  • this

    • this要在执行时才能确认值，定义时无法确认

      var a = {
    name: 'A',
    fn: function () {
        console.log(this.name);
    }
};
a.fn(); // this === a
a.fn.call({name: 'B'}); // this === {name: 'B'}
var fn1 = a.fn;
fn1(); // this === window

    • 作为构造函数执行

      function Foo(name){
    this.name = name;
}
var f = new Foo('KinW');

    • 作为对象属性执行

      var obj = {
    name: 'A',
    printName: function () {
        console.log(this.name);
    }
obj.printName();

    • 作为普通函数执行

      function fn() {
    console.log(this); // this === window    
}

    • call（最常用）、apply、bind

      function fn1(name) {
    alert(name);
    console.log(this);
}
fn1.call({x:100},'KinW',20);
fn1.apply({x:100},['KinW',20]);

var fn2 =function (name,age)
{
    alert(name);
    console.log(this);
}.bind({y:200})
fn2('KinW',20)

  • 作用域

    • JavaScript中没有块级作用域（不要在块内声明变量，不易读）

      // 无块级作用域
if (true){
    var name = 'KinW';
}
console.log(name); // 可以读取块内变量

    • 只有函数和全局作用域

      // 函数和全局作用域
var a = 100;
function fn(){
    var a = 200;
    console.log('fn',a);
}
console.log('global',a);
fn();

    • 作用域链

      var a = 100;
function fn(){
    var b = 200;

    // 当前作用域没有定义的变量，即“自由变量”
    console.log(a); // 函数的父级作用域 -> 全局作用域
    console.log(b);
}
fn();

var a =100;
function F1(){
    var b = 200;
    function F2(){
        var c = 300;
        console.log(a); // a 是自由变量
        console.log(b); // b 是自由变量
        console.log(c);
    }    
    F2();
}
F1();
